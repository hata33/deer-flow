```python
# scripts/analyze.py
import sys
import json
import argparse
from typing import Dict, Any

def main():
    """主入口函数"""
    # 解析输入
    parser = argparse.ArgumentParser()
    parser.add_argument('--input', required=True, help='用户输入')
    parser.add_argument('--context', help='上下文信息')
    args = parser.parse_args()

    # 执行分析
    try:
        result = analyze_stock(args.input, args.context)
        print(json.dumps(result, ensure_ascii=False, indent=2))
    except Exception as e:
        error_result = {
            "success": False,
            "error": str(e)
        }
        print(json.dumps(error_result))

def analyze_stock(user_input: str, context: str = None) -> Dict[str, Any]:
    """
    分析股票

    Args:
        user_input: 用户输入，如"分析腾讯控股"
        context: 上下文信息

    Returns:
        分析结果
    """
    # 1. 解析股票信息
    stock_info = parse_stock_info(user_input)

    # 2. 获取股票数据
    stock_data = get_stock_data(stock_info['symbol'])

    # 3. 计算技术指标
    from scripts.indicators import calculate_indicators
    indicators = calculate_indicators(stock_data)

    # 4. 分析基本面
    fundamental = analyze_fundamental(stock_info)

    # 5. 分析技术面
    from scripts.indicators import analyze_technical
    technical = analyze_technical(indicators)

    # 6. 生成建议
    recommendation = generate_recommendation(fundamental, technical)

    # 7. 生成报告
    from scripts.report import generate_report
    report = generate_report(
        stock_info=stock_info,
        stock_data=stock_data,
        indicators=indicators,
        fundamental=fundamental,
        technical=technical,
        recommendation=recommendation
    )

    return {
        "success": True,
        "stock": stock_info,
        "data": stock_data,
        "indicators": indicators,
        "fundamental": fundamental,
        "technical": technical,
        "recommendation": recommendation,
        "report": report
    }

def parse_stock_info(user_input: str) -> Dict[str, Any]:
    """解析股票信息"""
    # 简化实现：从输入提取股票名称
    # 实际应该用NLP或规则匹配

    # 常见股票映射
    stock_map = {
        "腾讯": {"symbol": "00700.HK", "name": "腾讯控股"},
        "腾讯控股": {"symbol": "00700.HK", "name": "腾讯控股"},
        "茅台": {"symbol": "600519.SH", "name": "贵州茅台"},
        "贵州茅台": {"symbol": "600519.SH", "name": "贵州茅台"},
        "苹果": {"symbol": "AAPL", "name": "Apple Inc."},
        "Apple": {"symbol": "AAPL", "name": "Apple Inc."}

    for name, info in stock_map.items():

            return info

    # 如果没找到，返回输入作为股票名称
    return {
        "symbol": user_input,
        "name": user_input
    }

def analyze_fundamental(stock_info: Dict) -> Dict[str, Any]:

    return {
        "score": 75,
        "analysis": "公司基本面良好，盈利能力稳定，行业地位突出",
        "strengths": ["行业龙头", "现金流充沛", "盈利能力强"],
        "weaknesses": ["估值偏高", "增长放缓"]

def generate_recommendation(fundamental: Dict, technical: Dict) -> Dict[str, Any]:
    """生成投资建议"""
    # 综合基本面和技术面
    fund_score = fundamental.get('score', 50)
    tech_score = technical.get('score', 50)

    avg_score = (fund_score + tech_score) / 2

    if avg_score >= 80:
        action = "强烈买入"
        risk = "低"
    elif avg_score >= 60:
        action = "买入"
        risk = "中"
    elif avg_score >= 40:
        action = "持有"
        risk = "中"
    else:
        action = "卖出"
        risk = "高"

    return {
        "action": action,
        "score": avg_score,
        "risk_level": risk,
        "reasoning": f"基本面评分{fund_score}，技术面评分{tech_score}，综合建议{action}"
    }

# scripts/indicators.py
from typing import Dict, List

def calculate_indicators(stock_data: Dict) -> Dict[str, Any]:
    """计算技术指标"""
    price = stock_data['current_price']

    # 模拟计算（实际应该用历史数据）
    return {
        "MA": {
            "MA5": price * 0.98,
            "MA10": price * 0.97,
            "MA20": price * 0.95,
            "MA60": price * 0.92
        },
        "MACD": {
            "dif": 2.5,
            "dea": 2.0,
            "histogram": 0.5,
            "signal": "买入"
        },
        "RSI": {
            "value": 55,
            "signal": "中性"
        },
        "KDJ": {
            "k": 60,
            "d": 55,
            "j": 70,
            "signal": "偏强"
        }
    }

def analyze_technical(indicators: Dict) -> Dict[str, Any]:
    """分析技术面"""
    score = 0
    signals = []

    # MA分析
    ma = indicators['MA']
    if ma['MA5'] > ma['MA20']:
        score += 20
        signals.append("短期均线向上，趋势良好")

    # MACD分析
    macd = indicators['MACD']
    if macd['histogram'] > 0:
        score += 20
        signals.append("MACD金叉，动能增强")

    # RSI分析
    rsi = indicators['RSI']['value']
    if 40 < rsi < 60:
        score += 15
        signals.append("RSI中性，观望为主")

    # KDJ分析
    kdj = indicators['KDJ']
    if kdj['k'] > kdj['d']:
        score += 15
        signals.append("KDJ向上，短期偏强")

    return {
        "score": score + 30,  # 基础分30
        "signals": signals,
        "summary": "技术面整体偏多，注意风险控制"
    }
```

### 4.3 报告生成

```python
# scripts/report.py
from typing import Dict, Any

def generate_report(**kwargs) -> str:
    """生成分析报告"""
    stock_info = kwargs['stock_info']
    stock_data = kwargs['stock_data']
    recommendation = kwargs['recommendation']

    report = f"""
# {stock_info['name']}（{stock_info['symbol']}）分析报告

## 基本信息
- 当前价格：{stock_data['current_price']:.2f}
- 涨跌幅：{stock_data['change_percent']:.2f}%
- 成交量：{stock_data['volume']:,}
- 市值：{stock_data['market_cap'] / 1e12:.2f}万亿

## 技术指标分析
{format_indicators(kwargs['indicators'])}

## 技术面分析
{format_technical(kwargs['technical'])}

## 基本面分析
{format_fundamental(kwargs['fundamental'])}

## 投资建议
**{recommendation['action']}**
- 综合评分：{recommendation['score']:.0f}/100
- 风险等级：{recommendation['risk_level']}
- 建议：{recommendation['reasoning']}

---
*本报告仅供参考，投资有风险，入市需谨慎*
"""

    return report.strip()

def format_indicators(indicators: Dict) -> str:
    """格式化技术指标"""
    lines = []
    for name, values in indicators.items():
        lines.append(f"- **{name}**: {values}")
    return "\n".join(lines)

def format_technical(technical: Dict) -> str:
    """格式化技术面"""
    return f"""
评分：{technical['score']}/100
信号：
{chr(10).join(f'- {s}' for s in technical['signals'])}
总结：{technical['summary']}
"""

def format_fundamental(fundamental: Dict) -> str:
    """格式化基本面"""
    return f"""
评分：{fundamental['score']}/100
优势：{', '.join(fundamental['strengths'])}
劣势：{', '.join(fundamental['weaknesses'])}
分析：{fundamental['analysis']}
"""
```

## 10. 最佳实践

### 10.1 开发原则

```
1. 单一职责
   → 每个技能做一件事
   → 做到极致

2. 幂等性
   → 相同输入产生相同输出
   → 便于测试和调试

3. 错误处理
   → 优雅处理错误
   → 给用户明确提示

4. 日志记录
   → 记录关键操作
   → 便于问题排查

5. 性能考虑
   → 避免重复计算
   → 使用缓存
```
